286  /*
287  When sampling w/o replacement, the new population size cannot be so large
288  that the number of individuals required from deme i or j is >= than those
289  demes'
290  respective sizes. This test samples so many individuals that this
291  condition cannot be met,
292  resulting in a return value of 1
293  */
294 
295  {
296  diploids = { dipvector_t(1000, std::make_pair(0, 0)),
297  dipvector_t(1000, std::make_pair(1, 1)) };
298  gametes = { KTfwd::gamete(2000), KTfwd::gamete(2000) };
299  // here, we want a new deme that is 50% ancestry from demes 0 and 1.
300  // But, sampling w/o
301  // replacement for 2000 individuals will require all individuals from
302  // demes i and j.
303  auto rv = KTfwd::admix_demes(rng_ptr, mutations, mcounts, gametes,
304  diploids, 0, 1, 0.5, 2000, false);
305  BOOST_REQUIRE_EQUAL(rv, 1);
306  }
