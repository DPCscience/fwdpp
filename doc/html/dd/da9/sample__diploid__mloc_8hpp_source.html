34  typename fitness_func>
35  void
36  w(const dipcont_t &diploids, gcont_t &gametes,
37  const mcont_t &mutations, const fitness_func &ff)
38  {
39  unsigned N_curr = diploids.size();
40  if (fitnesses.size() < N_curr)
41  fitnesses.resize(N_curr);
42  wbar = 0.;
43 
44  for (unsigned i = 0; i < N_curr; ++i)
45  {
46  for (auto region : diploids[i])
47  {
48  gametes[region.first].n
49  = gametes[region.second].n = 0;
50  }
51  // No need to dispatch--this type of simulation is
52  // always
53  // regarded as a non-standard diploid type.
54  fitnesses[i] = ff(diploids[i], gametes, mutations);
55  }
56 
57  wbar /= double(diploids.size());

60  Black magic alert:
61  fwdpp_internal::gsl_ran_discrete_t_ptr contains a
62  std::unique_ptr wrapping the GSL pointer.
63  This type has its own deleter, which is convenient, because
64  operator= for unique_ptrs automagically calls the deleter
65  before assignment!
66  Details:
67  http://www.cplusplus.com/reference/memory/unique_ptr/operator=

69  This only works b/c the rhs of the expression below may be
70  treated as an rvalue reference.
71  */
72  lookup = fwdpp_internal::gsl_ran_discrete_t_ptr(
73  gsl_ran_discrete_preproc(N_curr, &fitnesses[0]));
74  }
