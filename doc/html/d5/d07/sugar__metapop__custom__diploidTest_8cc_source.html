53 simulate(poptype &pop)
54 {
55  // Evolve for 10 generations
56  std::vector<std::function<double(const poptype::diploid_t &,
57  const poptype::gcont_t &,
58  const poptype::mcont_t &)>>
59  fitness_funcs(2, [](const poptype::diploid_t &d,
60  const poptype::gcont_t &g,
61  const poptype::mcont_t &m) {
62  return KTfwd::multiplicative_diploid()(g[d.first], g[d.second], m);
63  });
64  KTfwd::GSLrng_t<KTfwd::GSL_RNG_TAUS2> rng(0u);
65  for (unsigned generation = 0; generation < 10; ++generation)
66  {
67  std::vector<double> wbar = KTfwd::sample_diploid(
68  rng.get(), pop.gametes, pop.diploids, pop.mutations,
69  pop.mcounts, &pop.Ns[0], 0.005,
70  std::bind(KTfwd::infsites(), std::placeholders::_1,
71  std::placeholders::_2, rng.get(),
72  std::ref(pop.mut_lookup), generation, 0.005, 0.,
73  [&rng]() { return gsl_rng_uniform(rng.get()); },
74  []() { return 0.; }, []() { return 0.; }),
75  std::bind(KTfwd::poisson_xover(), rng.get(), 0.005, 0., 1.,
76  std::placeholders::_1, std::placeholders::_2,
77  std::placeholders::_3),
78  fitness_funcs,
79  std::bind(migpop, std::placeholders::_1, rng.get(), 0.001),
80  pop.neutral, pop.selected);
81  KTfwd::update_mutations(pop.mutations, pop.fixations,
82  pop.fixation_times, pop.mut_lookup,

84  }
