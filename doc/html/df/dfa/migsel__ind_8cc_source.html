57 // fitness model details -- s will be treated as -s in population 2
58 struct multiplicative_diploid_minus
59 {
60  typedef double result_type;
61  inline double
62  operator()(const poptype::diploid_t &dip, const poptype::gcont_t &gametes,
63  const poptype::mcont_t &mutations,
64  const double scaling = 1.) const
65  {
66  using mut_t = poptype::mcont_t::value_type;
67  return site_dependent_fitness()(
68  gametes[dip.first], gametes[dip.second], mutations,
69  [&](double &fitness, const mut_t &mut) {
70  fitness *= (1. - scaling * mut.s);
71  },
72  [](double &fitness, const mut_t &mut) {
73  fitness *= (1. - mut.h * mut.s);
74  },
75  1.);
76  }
77 };
