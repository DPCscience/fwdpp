152  /*!
153  Return a KTfwd::popgenmut
154 
155  \param r A gsl_rng
156  \param nmu Neutral mutation rate (per gamete, per generation)
157  \param smu Selected mutation rate (per gamete, per
158  generation)
159  \param generation The current generation
160  \param recycling_bin A recycling bin for mutations
161  \param mutations A container of mutations
162  \param lookup Lookup table for mutations
163 
164  \precondition If nmu == 0, then nbeg/nend cannot be empty.
165  Similarly,
166  if smu == 0, then sbeg,end cannot be empty. These conditions
167  are
168  checked in debug
169  mode via the assert macro. It is up to the calling
170  environment
171  to prevent this situation
172  from arising. Also, nmu+smu must be > 0, and is also checked
173  by
174  assert.
175  */
176  template <typename queue_t, typename lookup_table_t,
177  typename mcont_t>
178  inline result_type
179  make_mut(queue_t &recycling_bin, mcont_t &mutations,
180  const gsl_rng *r, const double nmu, const double smu,
181  const unsigned *generation, lookup_table_t &lookup) const
182  {
183  assert(nmu + smu > 0.);
184  bool is_neutral
185  = (gsl_rng_uniform(r) < nmu / (nmu + smu)) ? true : false;
186  if (is_neutral)
187  {
188  assert(!data->nbeg.empty());
189  assert(!data->nend.empty());
190  size_t region = gsl_ran_discrete(r, nlookup.get());
191  double pos = posmaker(r, data->nbeg[region],
192  data->nend[region], lookup);
193  return fwdpp_internal::recycle_mutation_helper(
194  recycling_bin, mutations, pos, 0., 0., *generation,
195  data->nlabels[region]);
196  }
197  assert(!data->sbeg.empty());
198  assert(!data->send.empty());
199  size_t region = gsl_ran_discrete(r, slookup.get());
200  double pos = posmaker(r, data->sbeg[region],
201  data->send[region], lookup);
202  return fwdpp_internal::recycle_mutation_helper(
203  recycling_bin, mutations, pos, data->shmodels[region].s(r),
204  data->shmodels[region].h(r), *generation,
205  data->slabels[region]);
206  }

326  /*!
327  Returns a position from a region that is chosen based on
328  region
329  weights.
330 
331  \precondition If recrate == 0, then beg and end cannot be
332  empty.
333  It is up to the calling environment to make sure
334  that this cannot happen. This is checked in debug mode via
335  the
336  assert macro.
337  */
338  template <typename gamete_t, typename mcont_t>
339  inline result_type
340  operator()(const gsl_rng *r, const double recrate,
341  const gamete_t &, const gamete_t &,
342  const mcont_t &) const
343  {
344  assert(!(recrate == 0.
345  && (data->beg.empty() || data->end.empty())));
346  auto nbreaks = gsl_ran_poisson(r, recrate);
347  if (!nbreaks)
348  return {};
349 
350  result_type rv;
351  for (unsigned i = 0; i < nbreaks; ++i)
352  {
353  size_t region = gsl_ran_discrete(r, lookup.get());
354  rv.push_back(gsl_ran_flat(r, data->beg[region],
355  data->end[region]));
356  }
357  std::sort(rv.begin(), rv.end());
358  rv.push_back(std::numeric_limits<double>::max());
359  return rv;
360  }
361  };
